    $
    \textrm{MCC} =
        \frac{
            n\left(\sum_{i=1}^{k}n_{ii}\right) -
            \left(\sum_{i=1}^{k}n_{i\cdot}n_{\cdot{}i}\right)
        }{
            \sqrt{
                \left(n^2-\sum_{i=1}^{k}n_{i\cdot}^2\right)
                \left(n^2-\sum_{i=1}^{k}n_{\cdot{}i}^2\right)
            }
        }
        \in{} [-1, 1]
    $
